class Solution:
    def mySqrt(self, x: int) -> int:
        a = x
        xk = 1
        epsilon = 1e-5
        while True:
            xk_1 = xk - (xk*xk - a)/(2.0*xk)
            if abs(xk_1-xk) < epsilon:
                break
            xk = xk_1
        return int(xk)
